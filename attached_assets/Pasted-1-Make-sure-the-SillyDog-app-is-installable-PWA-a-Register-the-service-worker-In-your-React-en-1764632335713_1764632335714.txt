c) Add an in-app ‚ÄúInstall SillyDog App‚Äù banner

Create a new component at
client/src/components/DownloadAppBanner.tsx:

import React, { useEffect, useState } from "react";

export const DownloadAppBanner: React.FC = () => {
  const [deferredPrompt, setDeferredPrompt] = useState<any>(null);
  const [canInstall, setCanInstall] = useState(false);

  useEffect(() => {
    const handler = (e: any) => {
      e.preventDefault();
      setDeferredPrompt(e);
      setCanInstall(true);
    };

    window.addEventListener("beforeinstallprompt", handler);

    return () => window.removeEventListener("beforeinstallprompt", handler);
  }, []);

  const handleInstallClick = async () => {
    if (!deferredPrompt) return;

    deferredPrompt.prompt();
    const { outcome } = await deferredPrompt.userChoice;
    console.log("User response to install prompt:", outcome);

    setDeferredPrompt(null);
    setCanInstall(false);
  };

  if (!canInstall) return null;

  return (
    <div
      style={{
        padding: "0.75rem 1rem",
        borderTop: "1px solid #e5e7eb",
        background: "#f9fafb",
        display: "flex",
        justifyContent: "space-between",
        alignItems: "center",
        gap: "0.75rem",
      }}
    >
      <span>Install the SillyDog app for faster booking on your phone.</span>
      <button
        onClick={handleInstallClick}
        style={{
          padding: "0.5rem 1rem",
          borderRadius: "999px",
          border: "none",
          cursor: "pointer",
        }}
      >
        Install app
      </button>
    </div>
  );
};


Then include it somewhere global, like in client/src/App.tsx (or your main layout component):

import { DownloadAppBanner } from "./components/DownloadAppBanner";

function App() {
  return (
    <>
      {/* your existing layout / router / pages here */}
      {/* ... */}
      <DownloadAppBanner />
    </>
  );
}

export default App;


On Android Chrome / desktop Chrome, users will now get an ‚ÄúInstall app‚Äù button when the browser says the PWA is installable.
